```java
class Tree {
    int x;
    int y;
    // 상하좌우에 나무가 있는 지역의 개수
    int treeCount;
    // 상하좌우에 나무가 없는 지역 방향 저장
    List<Integer> emptyDir;

    public Tree(int x, int y) {
        this.x = x;
        this.y = y;
    }

    void setTreeCount(int treeCount) {
        this.treeCount = treeCount;
    }

    void setEmptyDir(List<Integer> emptyDir) {
        this.emptyDir = emptyDir;
    }
}
```
